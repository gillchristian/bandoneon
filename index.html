JavaScript

// --- on document ready ---
$(function() {

// --- simple init ---
// --- default options ---
$('.container').jcorddion();

// --- nested jcorddion ---
// --- make sure to have different heading and content selectors ---
// --- custom options ---
$('.nested').jcorddion({
		toggle: '.nested-jcorddion-heading',
		content: '.nested-jcorddion-content',
		event: 'hover',
		opened: '#first', 
		speed: 'slow'
});

// --- default options ---
{
	toggle: '.jcorddion-heading', // heading selector
	content: '.jcorddion-content', // content selector
	opened: null, // the one that starts open
	event: 'click', // event that triggers the toggle
	speed: 'fast' // toggle speed
}

});
